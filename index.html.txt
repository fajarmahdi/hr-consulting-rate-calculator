    <script>
        function formatIDR(num) {
            if (num === 0) return "IDR 0";
            if (num >= 1000000000) return "IDR " + (num / 1000000000).toFixed(2) + " Billion";
            if (num >= 1000000) return "IDR " + (num / 1000000).toFixed(1) + " Million";
            return "IDR " + num.toLocaleString('id-ID');
        }

        function handleInputChain() {
            const revenue = parseFloat(document.getElementById('revenue').value) || 0;
            const gop = parseFloat(document.getElementById('gop').value) || 0;
            const gmSalary = parseFloat(document.getElementById('gmSalary').value) || 0;
            const headcount = parseFloat(document.getElementById('headcount').value) || 0;

            // Sequential Unlock (Visual Only - logic runs regardless)
            if (revenue > 0) document.getElementById('group_gop').classList.remove('disabled');
            else document.getElementById('group_gop').classList.add('disabled');

            if (gop > 0) document.getElementById('group_gm').classList.remove('disabled');
            else document.getElementById('group_gm').classList.add('disabled');

            if (gmSalary > 0) document.getElementById('group_headcount').classList.remove('disabled');
            else document.getElementById('group_headcount').classList.add('disabled');

            if (headcount > 0) document.getElementById('group_visits').classList.remove('disabled');
            else document.getElementById('group_visits').classList.add('disabled');

            calculate();
        }

        function calculate() {
            const revenue = parseFloat(document.getElementById('revenue').value) || 0;
            const gop = parseFloat(document.getElementById('gop').value) || 0;
            const gmSalary = parseFloat(document.getElementById('gmSalary').value) || 0;
            const headcount = parseFloat(document.getElementById('headcount').value) || 0;
            const visitDays = parseFloat(document.getElementById('visitDays').value) || 0;

            // --- 1. DETERMINE BASE FEE (The "Running Meter" Logic) ---
            // Priority: GM Salary (Anchor) > GOP (Profit Proxy) > Revenue (Scale Proxy)
            
            let base_fee = 0;
            let logic_msg = "Waiting...";
            let alert_msg = "";
            let cap_hit = false;

            if (gmSalary > 0) {
                // Tier 1: Precise Anchor (60% of GM)
                base_fee = gmSalary * 0.60;
                logic_msg = `Based on 60% of GM Salary.`;
            } else if (gop > 0) {
                // Tier 2: Profit Proxy (1.5% of GOP)
                base_fee = gop * 0.015;
                logic_msg = `Estimated at 1.5% of Monthly Profit.`;
            } else if (revenue > 0) {
                // Tier 3: Revenue Proxy (0.4% of Revenue)
                base_fee = revenue * 0.004;
                logic_msg = `Estimated at 0.4% of Revenue.`;
            } else {
                // Reset
                resetOutputs();
                return;
            }

            // --- 2. APPLY CAPS (The "Influence" Check) ---
            // If GOP is known, capped at 10% GOP (Affordability)
            if (gop > 0) {
                const profit_cap = gop * 0.10;
                if (base_fee > profit_cap) {
                    base_fee = profit_cap;
                    cap_hit = true;
                    logic_msg += " (Capped at 10% Profit)";
                }
            }

            // Rounding for cleanliness
            base_fee = Math.round(base_fee / 500000) * 500000;
            if (base_fee < 0) base_fee = 0;


            // --- 3. OUTPUT OPTION A (Pure Contract) ---
            document.getElementById('optA_price').innerText = formatIDR(base_fee) + " / mo";
            document.getElementById('optA_logic').innerText = logic_msg;
            
            let alertElem = document.getElementById('optA_alert');
            if (cap_hit) {
                alertElem.innerText = "⚠️ Fee reduced by Profit Cap.";
                alertElem.style.display = 'block';
            } else {
                alertElem.style.display = 'none';
            }


            // --- 4. OUTPUT OPTION B (Cash + Credit) ---
            let cash_portion = Math.round((base_fee * 0.70) / 500000) * 500000;
            let credit_val = (base_fee - cash_portion) * 1.5;
            let credit_portion = Math.round(credit_val / 500000) * 500000;

            document.getElementById('optB_price').innerText = formatIDR(cash_portion + credit_portion) + " Value";
            document.getElementById('optB_breakdown').innerText = `${formatIDR(cash_portion)} Cash + ${formatIDR(credit_portion)} Credit`;
            document.getElementById('optB_logic').innerText = `70% Cash / 30% Credit (1.5x Multiplier)`;


            // --- 5. OUTPUT OPTION C (Variable) ---
            // Floor: 50% of Base Fee. Visit Rate: 10% of Base Fee per day.
            let c_floor = Math.round((base_fee * 0.50) / 500000) * 500000;
            let c_daily = Math.round((base_fee * 0.10) / 100000) * 100000;
            let c_total = c_floor + (c_daily * visitDays);

            document.getElementById('optC_price').innerText = formatIDR(c_total) + " / mo (Est)";
            document.getElementById('optC_logic').innerText = `Base ${formatIDR(c_floor)} + ${visitDays} Visits @ ${formatIDR(c_daily)}`;


            // --- 6. OUTPUT OPTION D (Headcount) ---
            // If headcount exists, calc rate. If not, estimate based on Revenue/GM proxies.
            let d_val = 0;
            let d_msg = "Waiting for headcount...";
            
            // Rate per head logic: 
            // If GM Salary known: 0.22% of GM Salary per head
            // If only Fee known: Fee / 200 (Est headcount proxy)
            
            let rate_per_head = 0;
            
            if (gmSalary > 0) {
                rate_per_head = gmSalary * 0.0022;
            } else {
                // Fallback: Estimate rate based on total base_fee (assuming ~200 pax scale)
                rate_per_head = base_fee / 180; 
            }

            // Clean round rate
            rate_per_head = Math.round(rate_per_head / 5000) * 5000; 
            if (rate_per_head < 50000) rate_per_head = 50000; // Minimum 50k floor

            if (headcount > 0) {
                d_val = headcount * rate_per_head;
                d_msg = `${headcount} pax @ ${formatIDR(rate_per_head)}/head`;
            } else {
                // Show estimated rate even if headcount not entered
                d_msg = `Est. Rate: ${formatIDR(rate_per_head)}/head (Enter count)`;
            }

            d_val = Math.round(d_val / 500000) * 500000;
            document.getElementById('optD_price').innerText = formatIDR(d_val) + " / mo";
            document.getElementById('optD_logic').innerText = d_msg;
        }

        function resetOutputs() {
            document.getElementById('optA_price').innerText = "IDR 0";
            document.getElementById('optA_logic').innerText = "Waiting for inputs...";
            document.getElementById('optB_price').innerText = "IDR 0";
            document.getElementById('optC_price').innerText = "IDR 0";
            document.getElementById('optD_price').innerText = "IDR 0";
        }
